<script>

    // Load from LocalStorage

    let queue = JSON.parse(localStorage.getItem('pb_queue')) || [];

    let history = JSON.parse(localStorage.getItem('pb_history')) || [];

    let groupID = localStorage.getItem('pb_id') ? parseInt(localStorage.getItem('pb_id')) : 1;


    function save() {

        localStorage.setItem('pb_queue', JSON.stringify(queue));

        localStorage.setItem('pb_history', JSON.stringify(history));

        localStorage.setItem('pb_id', groupID);

        render();

    }


    function add(size) {

        queue.push({ 

            id: groupID++, 

            size: size, 

            time: new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}) 

        });

        save();

    }


    function nextMatch() {

        if (queue.length === 0) return alert("Rack is currently empty.");

        

        let matchPlayers = 0;

        let matchGroups = [];


        // Pull players until we hit 4 or the next group is too big

        while (queue.length > 0 && (matchPlayers + queue[0].size <= 4)) {

            let group = queue.shift();

            matchPlayers += group.size;

            matchGroups.push(`#${group.id}`);

        }


        if (matchGroups.length > 0) {

            const timestamp = new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'});

            const entry = `${timestamp} | Groups: ${matchGroups.join(', ')} (${matchPlayers}P)`;

            history.unshift(entry);

            if (history.length > 5) history.pop();

            save();

        } else {

            alert("The next group in line is too large for the remaining spots.");

        }

    }


    function render() {

        const rackEl = document.getElementById('rack');

        rackEl.innerHTML = queue.length ? '' : '<p style="text-align:center;color:#ccc;padding-top:60px; font-weight:bold;">NO PADDLES STACKED</p>';

        queue.forEach(g => {

            rackEl.innerHTML += `<div class="group-card">

                <span><b style="color:var(--primary-orange)">GROUP #${g.id}</b> <br><small style="color:#999">${g.time}</small></span>

                <span style="font-weight: 800; font-size: 1.2rem;">${g.size}P</span>

            </div>`;

        });


        const logEl = document.getElementById('history-log');

        logEl.innerHTML = history.map(h => `<div class="history-item">${h}</div>`).join('');

    }


    function clearData() {

        if(confirm("Confirm: This will delete all current queue data.")) {

            localStorage.clear();

            queue = []; history = []; groupID = 1;

            save();

        }

    }


    // Auto-generate QR code for the current URL

    new QRCode(document.getElementById("qrcode"), {

        text: window.location.href,

        width: 140,

        height: 140,

        colorDark : "#333333",

        colorLight : "#ffffff"

    });


    render();

</script>
